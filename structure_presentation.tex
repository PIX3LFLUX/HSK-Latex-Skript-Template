\newcommand{\placetextbox}[4]{%
  % [#1]: box anchor: center | 
  %                 south west | west | north west (default) | north |
  %                 north east | east | south east | south | 
  %                 mid west | mid | mid east |
  %                 base west | base | base east 
  % #2: horizontal position
  % #3: vertical position
  % #4: maximum width
  % #5: content
  %
    \begin{tikzpicture}[remember picture, overlay]
    \node[inner sep=0pt, text width=#3, anchor=north west] at ($(current page.north west) + (#1, #2)$)
    {   
        #4
    };            
    \end{tikzpicture}
}
